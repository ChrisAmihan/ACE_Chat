Create DataBase project2;
USE project2;

Create Table User_Table
(User_Id int unsigned not null auto_increment Primary key,
Username varchar(50) unique not null,
Name varchar (50) not null,
Profile_Pic varchar(500)
);

Create Table Friend_Table
(Friend_Table_Id int unsigned not null auto_Increment primary key,
Requester_Id int unsigned not null,
Requestee_Id int unsigned not null,
Status varchar(15),
Foreign key (Requester_Id) references User_Table(User_Id),
Foreign key (Requestee_Id) references User_Table(User_Id)
);

Create Table Block_User_Table
(Block_User_Table_Id int unsigned not null auto_Increment primary key,
Blocker_Id int unsigned not null,
Blockee_Id int unsigned not null,
Foreign key (Blocker_Id) references User_Table(User_Id),
Foreign key (Blockee_Id) references User_Table(User_Id)
)

Create table Chat_Table
(Chat_Id int unsigned not null auto_Increment primary key,
User_Id int unsigned not null,
Status varchar(10),
Foreign key (User_Id) references User_Table(User_Id)
);

Create table Chat_Log_Table
(Chat_Log_Id int unsigned not null auto_Increment primary key,
User_Id int unsigned not null,
Message varchar(1000),
Sent_Time TIMESTAMP,
Chat_Id int unsigned not null,
Foreign key (User_Id) references User_Table(User_Id),
Foreign key (Chat_Id) references Chat_Table(Chat_Id)
);
